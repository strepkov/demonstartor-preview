\begin{lstlisting} [caption={Example of a valid test for Context Conditions.},label={lst:valid-test},captionpos=b]
package test.coco.valid.units;

component MyComponentMix {
    ports
        out Q(0kg: 100kg) out1;
        
    implementation Math {

      Q (0kg: 100kg) var1 = 10kg;
      if var1 <= 10kg
        out1 = 10kg;
      end

      if 10kg < var1
        out1 = 10kg;
      end
      
      Q (0kg: 100kg) var2;
      var2 = 10kg;
      
      if var1 > var2
        out1 = 10kg;
      end
      
      if 10m == 10m
        out1 = 10kg;
      end
      Q var3 = 1000m;
    }
}
\end{lstlisting}
